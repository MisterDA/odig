val (>|=) : 'a t -> ('a -> 'b) -> 'b t

Monadic bind.

val (<*>) : ('a -> 'b) t -> 'a t -> 'b t

f <*> (Some x) returns Some (f x) and f <*> None returns None.

val (<$>) : ('a -> 'b) -> 'a t -> 'b t

Like map.

val (<+>) : 'a t -> 'a t -> 'a t
