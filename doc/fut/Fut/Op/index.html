Module Fut.Op

Infix operators.

In a module to open.

Binding operators

val (>>=) : 'a t -> ('a -> 'b t) -> 'b t

f >>= fn is bind f fn.

val (>>|) : 'a t -> ('a -> 'b) -> 'b t

f >>| fn is map fn f.

val (>>&) : ('a, 'c) result -> ('a -> ('b, 'c) result) -> ('b, 'c) result

f >>& fn is rbind f fn.
