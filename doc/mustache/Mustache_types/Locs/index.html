Module Mustache_types.Locs

type loc = {
loc_start : Stdlib.Lexing.position;
loc_end : Stdlib.Lexing.position;
}
type desc =
| String of string
| Escaped of dotted_name
| Section of section
| Unescaped of dotted_name
| Partial of partial
| Inverted_section of section
| Concat of t list
| Comment of string
and section = {
name : dotted_name;
contents : t;
}
and partial = {
indent : int;
name : name;
contents : t option Stdlib.Lazy.t;
}
and t = {
loc : loc;
desc : desc;
}
