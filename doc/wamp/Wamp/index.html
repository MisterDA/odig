type 'a msg =
| Hello of {
realm : Uri.t;
details : 'a dict;
}
| Welcome of {
id : int;
details : 'a dict;
}
| Abort of {
details : 'a dict;
reason : Uri.t;
}
| Goodbye of {
details : 'a dict;
reason : Uri.t;
}
| Error of {
reqtype : int;
reqid : int;
details : 'a dict;
error : Uri.t;
args : 'a list;
kwArgs : 'a dict;
}
| Publish of {
reqid : int;
options : 'a dict;
topic : Uri.t;
args : 'a list;
kwArgs : 'a dict;
}
| Published of {
reqid : int;
id : int;
}
| Subscribe of {
reqid : int;
options : 'a dict;
topic : Uri.t;
}
| Subscribed of {
reqid : int;
id : int;
}
| Unsubscribe of {
reqid : int;
id : int;
}
| Unsubscribed of int
| Event of {
subid : int;
pubid : int;
details : 'a dict;
args : 'a list;
kwArgs : 'a dict;
}

val unsubscribed : reqid:int -> 'a msg
